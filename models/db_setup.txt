DROP TABLE IF EXISTS `games`;
		
CREATE TABLE `games` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `map` INTEGER NULL DEFAULT NULL,
  `totalTurns` INTEGER NULL DEFAULT NULL,
  `players` INTEGER NULL DEFAULT NULL,
  `currentPlayerTurn` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'Players(in-game)'
-- 
-- ---

DROP TABLE IF EXISTS `Players(in-game)`;
		
CREATE TABLE `Players(in-game)` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `income` INTEGER NULL DEFAULT NULL,
  `wallet` INTEGER NULL DEFAULT NULL,
  `character` INTEGER NULL DEFAULT NULL,
  `specialMeter` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'users'
-- 
-- ---

DROP TABLE IF EXISTS `users`;
		
CREATE TABLE `users` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `name` INTEGER NULL DEFAULT NULL,
  `email` INTEGER NULL DEFAULT NULL,
  `password` INTEGER NULL DEFAULT NULL,
  `avatar` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
KEY (),
KEY ()
);

-- ---
-- Table 'units(in-game)'
-- 
-- ---

DROP TABLE IF EXISTS `units(in-game)`;
		
CREATE TABLE `units(in-game)` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `unitID` INTEGER NULL DEFAULT NULL,
  `posX` INTEGER NULL DEFAULT NULL,
  `posY` INTEGER NULL DEFAULT NULL,
  `health` INTEGER NULL DEFAULT NULL,
  `playerID` INTEGER NULL DEFAULT NULL,
  `terrain` INTEGER NULL DEFAULT NULL,
  `type` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'buildingTypes'
-- 
-- ---

DROP TABLE IF EXISTS `buildingTypes`;
		
CREATE TABLE `buildingTypes` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `sprite` INTEGER NULL DEFAULT NULL,
  `type(factory,base,city)` INTEGER NULL DEFAULT NULL,
  `income` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'chat'
-- 
-- ---

DROP TABLE IF EXISTS `chat`;
		
CREATE TABLE `chat` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `userID` INTEGER NULL DEFAULT NULL,
  `message` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'maps'
-- 
-- ---

DROP TABLE IF EXISTS `maps`;
		
CREATE TABLE `maps` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `name` INTEGER NULL DEFAULT NULL,
  `numPlayers` INTEGER NULL DEFAULT NULL,
  `mapImage` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'characters'
-- 
-- ---

DROP TABLE IF EXISTS `characters`;
		
CREATE TABLE `characters` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `name` INTEGER NULL DEFAULT NULL,
  `passive` INTEGER NULL DEFAULT NULL,
  `sprite` INTEGER NULL DEFAULT NULL,
  `special` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'unitTypes'
-- 
-- ---

DROP TABLE IF EXISTS `unitTypes`;
		
CREATE TABLE `unitTypes` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `type(infantry,tank,plane,etc)` INTEGER NULL DEFAULT NULL,
  `sprite` INTEGER NULL DEFAULT NULL,
  `cost` INTEGER NULL DEFAULT NULL,
  `damage` INTEGER NULL DEFAULT NULL,
  `health` INTEGER NULL DEFAULT NULL,
  `range` INTEGER NULL DEFAULT NULL,
  `speed(movement)` INTEGER NULL DEFAULT NULL,
  `movementType` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'terrainTypes'
-- 
-- ---

DROP TABLE IF EXISTS `terrainTypes`;
		
CREATE TABLE `terrainTypes` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `name` INTEGER NULL DEFAULT NULL,
  `defenseBonus` INTEGER NULL DEFAULT NULL,
  `pathingType(0-X)` INTEGER NULL DEFAULT NULL,
  `moveChange` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'stats'
-- 
-- ---

DROP TABLE IF EXISTS `stats`;
		
CREATE TABLE `stats` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `wins` INTEGER NULL DEFAULT NULL,
  `losses` INTEGER NULL DEFAULT NULL,
  `totalMatches` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'mapTerrain'
-- 
-- ---

DROP TABLE IF EXISTS `mapTerrain`;
		
CREATE TABLE `mapTerrain` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `mapID` INTEGER NULL DEFAULT NULL,
  `type` INTEGER NULL DEFAULT NULL,
  `xCoord` INTEGER NULL DEFAULT NULL,
  `yCoord` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);

-- ---
-- Table 'buildings(in-game)'
-- 
-- ---

DROP TABLE IF EXISTS `buildings(in-game)`;
		
CREATE TABLE `buildings(in-game)` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `buildingID` INTEGER NULL DEFAULT NULL,
  `posX` INTEGER NULL DEFAULT NULL,
  `posY` INTEGER NULL DEFAULT NULL,
  `health` INTEGER NULL DEFAULT NULL,
  `owner` INTEGER NULL DEFAULT NULL,
  `type` INTEGER NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);
